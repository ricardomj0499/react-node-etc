# Cambiar x a letras en ujna oracion

# Hxla cxmx estax xicaxxo - ooosRrd

def replace_x(sentence, replace):
    print("Se van a cambiar las x de: %s, por los valores de: %s" % (sentence, replace))
    a = sentence
    for x in replace:
        print(x)
        a = a.replace("x", x, 1)
    return a
    
    
re = replace_x("Hxla cxmx estax xicaxxo", "ooosRrd")
print(re)
